CREATE DATABASE DESARROLLO
USE DESARROLLO
CREATE TABLE PERSONAS(
ID INT PRIMARY KEY IDENTITY(1,1),
DOCUMENTO VARCHAR(50),
NOMBRES VARCHAR(60),
APELLIDOS VARCHAR(60),
FECHANACIMIENTO DATE,
TIPO VARCHAR(10),
GENERO VARCHAR(10),
FECHAREGISTRO DATE,
FECHABAJA DATE,
USUARIO VARCHAR(150),
DIRECCION VARCHAR(200),
DESCRIPCION VARCHAR(500),
TELEFONOFIJO VARCHAR(20),
TELEFONOMOVIL VARCHAR(20),
EMAIL VARCHAR(150)
)
CREATE TABLE MAESTRAS(
ID INT IDENTITY(1,1) PRIMARY KEY,
TIPOMAESTRA VARCHAR(500)
)
CREATE TABLE DATAMAESTRA(
ID INT PRIMARY KEY IDENTITY(1,1),
NOMBRE VARCHAR (500),
ID_FK INT FOREIGN KEY REFERENCES MAESTRAS(ID)
)
